Google Apps Script Complete Deploy Guide

Highly Secure Database Integration

This script automates all CORS header routing, allowing instant REST interaction with 11 connected sheets in Google Drive.

Click the copy button below, navigate to your Google Sheets Apps Script panel, paste this exact block, click Save, deploy as a Web App (access to "Anyone"), and paste your web app URL inside Settings!

/**
 * ORPHANAGE MANAGEMENT SYSTEM - BACKEND DATABASE ENGINE (Google Apps Script)
 * Deployed as a Web App: Execute as "Me", Access: "Anyone"
 * 
 * This script serves as a robust JSON REST API for the frontend web application.
 * It manages 11 sheets within a single Google Spreadsheet, auto-creating them
 * with headers if they do not exist.
 */

// Define sheet names and their respective headers for auto-creation
const SHEET_SCHEMAS = {
  "Children": ["id", "name", "age", "gender", "entryDate", "guardianName", "guardianPhone", "roomNumber", "bedNumber", "status"],
  "Rooms": ["roomNumber", "capacity", "occupancy", "genderType", "notes"],
  "Food_Inventory": ["id", "itemName", "category", "stockIn", "stockOut", "currentStock", "unit", "expiryDate", "wastage", "notes"],
  "Daily_Meals": ["id", "date", "mealType", "plannedItems", "calories", "proteins", "carbs", "fats", "costPerChild", "notes"],
  "School_Enrollment": ["childId", "schoolName", "gradeClass", "rollNumber", "transportStatus"],
  "School_Fees": ["id", "childId", "amountDue", "amountPaid", "balance", "dueDate", "receiptUrl", "status"],
  "Academic_Reports": ["id", "childId", "term", "gradesJson", "teacherComments", "dateCreated"],
  "Medical_Records": ["childId", "bloodType", "allergies", "chronicConditions", "vaccinationsJson", "doctorVisitsJson", "dispensedMedsJson"],
  "Finances": ["id", "date", "type", "category", "amount", "description", "receiptUrl", "allocatedTo"],
  "Donations": ["id", "donorName", "donorEmail", "date", "amount", "donationType", "allocatedTo", "thanked", "receiptUrl"],
  "App_Settings": ["key", "value"]
};

function doGet(e) {
  const action = e.parameter.action;
  const output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);
  
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    if (action === "testConnection") {
      return responseJSON({ success: true, message: "Connected to Google Sheets successfully!", spreadsheetName: ss.getName() });
    }
    if (action === "setup") {
      setupSheets(ss);
      return responseJSON({ success: true, message: "All 11 sheets initialized successfully with correct headers!" });
    }
    if (action === "readAll") {
      setupSheets(ss);
      const data = {};
      for (let sheetName in SHEET_SCHEMAS) {
        const sheet = ss.getSheetByName(sheetName);
        data[sheetName.toLowerCase()] = getSheetData(sheet);
      }
      return responseJSON({ success: true, data: data });
    }
    return responseJSON({ success: false, error: "Invalid GET action." });
  } catch (err) {
    return responseJSON({ success: false, error: err.toString() });
  }
}

function doPost(e) {
  let postData;
  try {
    postData = JSON.parse(e.postData.contents);
  } catch (err) {
    return responseJSON({ success: false, error: "Invalid JSON body: " + err.toString() });
  }
  
  const action = postData.action;
  const sheetName = postData.sheetName;
  
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    setupSheets(ss);
    const sheet = ss.getSheetByName(sheetName);
    if (!sheet) return responseJSON({ success: false, error: "Sheet not found" });
    
    if (action === "saveRecord") {
      const record = postData.record;
      const keyColumn = postData.keyColumn || "id";
      const keyValue = record[keyColumn];
      
      const headers = SHEET_SCHEMAS[sheetName];
      const data = sheet.getDataRange().getValues();
      const keyIndex = headers.indexOf(keyColumn);
      
      let rowIndex = -1;
      for (let i = 1; i < data.length; i++) {
        if (data[i][keyIndex].toString() === keyValue.toString()) {
          rowIndex = i + 1;
          break;
        }
      }
      
      const rowValues = headers.map(header => {
        let val = record[header];
        if (val === undefined || val === null) return "";
        if (typeof val === "object") return JSON.stringify(val);
        return val;
      });
      
      if (rowIndex !== -1) {
        sheet.getRange(rowIndex, 1, 1, headers.length).setValues([rowValues]);
        return responseJSON({ success: true, message: "Record updated" });
      } else {
        sheet.appendRow(rowValues);
        return responseJSON({ success: true, message: "Record appended" });
      }
    }
    
    if (action === "deleteRecord") {
      const keyColumn = postData.keyColumn || "id";
      const keyValue = postData.keyValue;
      const headers = SHEET_SCHEMAS[sheetName];
      const data = sheet.getDataRange().getValues();
      const keyIndex = headers.indexOf(keyColumn);
      
      let rowIndex = -1;
      for (let i = 1; i < data.length; i++) {
        if (data[i][keyIndex].toString() === keyValue.toString()) {
          rowIndex = i + 1;
          break;
        }
      }
      
      if (rowIndex !== -1) {
        sheet.deleteRow(rowIndex);
        return responseJSON({ success: true, message: "Record deleted" });
      }
      return responseJSON({ success: false, error: "Record not found" });
    }
    
    return responseJSON({ success: false, error: "Invalid action" });
  } catch (err) {
    return responseJSON({ success: false, error: err.toString() });
  }
}

function setupSheets(ss) {
  for (let sheetName in SHEET_SCHEMAS) {
    let sheet = ss.getSheetByName(sheetName);
    const headers = SHEET_SCHEMAS[sheetName];
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      sheet.appendRow(headers);
      const range = sheet.getRange(1, 1, 1, headers.length);
      range.setFontWeight("bold").setBackground("#2c3e50").setFontColor("#ffffff");
      sheet.setFrozenRows(1);
    }
  }
}

function getSheetData(sheet) {
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];
  const headers = data[0];
  const rows = [];
  for (let i = 1; i < data.length; i++) {
    const row = {};
    let isEmpty = true;
    for (let j = 0; j < headers.length; j++) {
      let val = data[i][j];
      if (typeof val === "string" && (val.startsWith("[") || val.startsWith("{"))) {
        try { val = JSON.parse(val); } catch(e) {}
      }
      row[headers[j]] = val;
      if (val !== "") isEmpty = false;
    }
    if (!isEmpty) rows.push(row);
  }
  return rows;
}

function responseJSON(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
